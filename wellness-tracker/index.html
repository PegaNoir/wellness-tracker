  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <meta name="apple-mobile-web-app-status-bar-style" content="default">

  <meta name="apple-mobile-web-app-title" content="Wellness">

  <title>Wellness Tracker</title>

  <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

  <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

  <script src="https://cdn.tailwindcss.com"></script>

  <script src="https://unpkg.com/lucide@0.263.0/dist/umd/lucide.min.js"></script>

</head>

<body>

  <div id="root"></div>

  

  <script type="text/babel">

    const { useState, useEffect } = React;

    const { Calendar, Download, Settings, Plus, Trash2, Check, X, ChevronDown, ChevronUp, Bell, Cloud, CloudOff } = lucide;


    const WellnessTracker = () => {

      const [entries, setEntries] = useState([]);

      const [currentEntry, setCurrentEntry] = useState({});

      const [customAttributes, setCustomAttributes] = useState([]);

      const [showSettings, setShowSettings] = useState(false);

      const [newAttributeName, setNewAttributeName] = useState('');

      const [newAttributeType, setNewAttributeType] = useState('boolean');

      const [view, setView] = useState('today');

      const [googleSheetUrl, setGoogleSheetUrl] = useState('');

      const [isConnected, setIsConnected] = useState(false);

      const [isSyncing, setIsSyncing] = useState(false);

      const [notificationsEnabled, setNotificationsEnabled] = useState(false);

      const [expandedCategories, setExpandedCategories] = useState({

        sleep: true,

        recovery: false,

        physical: false,

        mental: false,

        nutrition: false,

        supplements: false,

        medications: false,

        lifestyle: false

      });


      const defaultAttributes = [

        { id: 'sleep_own_bed', name: 'Slept in own bed', type: 'boolean', category: 'sleep', enabled: true },

        { id: 'sleep_quality', name: 'Sleep quality', type: 'scale', category: 'sleep', enabled: true, min: 1, max: 10 },

        { id: 'energy_level', name: 'Energy level', type: 'scale', category: 'sleep', enabled: true, min: 1, max: 10 },

        { id: 'screen_time_bed', name: 'Screen time before bed', type: 'boolean', category: 'sleep', enabled: true },

        { id: 'podcast_sleeping', name: 'Podcast for sleeping', type: 'boolean', category: 'sleep', enabled: true },

        { id: 'sharing_bed', name: 'Sharing a bed', type: 'boolean', category: 'sleep', enabled: true },

        { id: 'bedtime_routine', name: 'Bedtime routine completed', type: 'boolean', category: 'sleep', enabled: true },

        { id: 'acupuncture', name: 'Acupuncture', type: 'boolean', category: 'recovery', enabled: true },

        { id: 'ice_bath', name: 'Ice bath', type: 'boolean', category: 'recovery', enabled: true },

        { id: 'massage', name: 'Massage', type: 'boolean', category: 'recovery', enabled: true },

        { id: 'stretching', name: 'Stretching', type: 'boolean', category: 'recovery', enabled: true },

        { id: 'physical_therapy', name: 'Physical therapy', type: 'boolean', category: 'recovery', enabled: true },

        { id: 'exercise', name: 'Exercised today', type: 'boolean', category: 'physical', enabled: true },

        { id: 'exercise_duration', name: 'Exercise duration (min)', type: 'number', category: 'physical', enabled: true },

        { id: 'physical_injury', name: 'Physical injury status', type: 'boolean', category: 'physical', enabled: true },

        { id: 'mood', name: 'Overall mood', type: 'scale', category: 'mental', enabled: true, min: 1, max: 10 },

        { id: 'stress_level', name: 'Stress level', type: 'scale', category: 'mental', enabled: true, min: 1, max: 10 },

        { id: 'meditation', name: 'Meditated today', type: 'boolean', category: 'mental', enabled: true },

        { id: 'meditation_duration', name: 'Meditation duration (min)', type: 'number', category: 'mental', enabled: true },

        { id: 'productivity', name: 'Productivity feeling', type: 'scale', category: 'mental', enabled: true, min: 1, max: 10 },

        { id: 'anxiety', name: 'Anxiety presence', type: 'boolean', category: 'mental', enabled: true },

        { id: 'therapy_session', name: 'Therapy session', type: 'boolean', category: 'mental', enabled: true },

        { id: 'journaling', name: 'Journaling (mental health)', type: 'boolean', category: 'mental', enabled: true },

        { id: 'alcohol', name: 'Consumed alcohol', type: 'boolean', category: 'nutrition', enabled: true },

        { id: 'caffeine', name: 'Consumed caffeine', type: 'boolean', category: 'nutrition', enabled: true },

        { id: 'caffeine_servings', name: 'Caffeine servings', type: 'number', category: 'nutrition', enabled: true },

        { id: 'water_intake', name: 'Water intake (glasses)', type: 'number', category: 'nutrition', enabled: true },

        { id: 'meals_quality', name: 'Meal quality', type: 'scale', category: 'nutrition', enabled: true, min: 1, max: 10 },

        { id: 'intermittent_fasting', name: 'Intermittent fasting', type: 'boolean', category: 'nutrition', enabled: true },

        { id: 'magnesium', name: 'Magnesium', type: 'boolean', category: 'supplements', enabled: true },

        { id: 'omega3', name: 'Omega-3', type: 'boolean', category: 'supplements', enabled: true },

        { id: 'vitamin_d', name: 'Vitamin D', type: 'boolean', category: 'supplements', enabled: true },

        { id: 'ibuprofen', name: 'Ibuprofen', type: 'boolean', category: 'medications', enabled: true },

        { id: 'social_interaction', name: 'Meaningful social interaction', type: 'boolean', category: 'lifestyle', enabled: true },

        { id: 'work_from_home', name: 'Working from home', type: 'boolean', category: 'lifestyle', enabled: true },

        { id: 'sexual_activity', name: 'Sexual activity', type: 'boolean', category: 'lifestyle', enabled: true },

      ];


      const [attributes, setAttributes] = useState(defaultAttributes);


      useEffect(() => {

        const savedEntries = localStorage.getItem('wellnessEntries');

        if (savedEntries) setEntries(JSON.parse(savedEntries));

        const savedAttrs = localStorage.getItem('customAttributes');

        if (savedAttrs) setCustomAttributes(JSON.parse(savedAttrs));

        const savedExpandedState = localStorage.getItem('expandedCategories');

        if (savedExpandedState) setExpandedCategories(JSON.parse(savedExpandedState));

        const savedEnabledState = localStorage.getItem('attributesEnabled');

        if (savedEnabledState) {

          const enabledState = JSON.parse(savedEnabledState);

          setAttributes(prev => prev.map(attr => ({

            ...attr,

            enabled: enabledState[attr.id] !== undefined ? enabledState[attr.id] : attr.enabled

          })));

        }

        const savedSheetUrl = localStorage.getItem('googleSheetUrl');

        if (savedSheetUrl) {

          setGoogleSheetUrl(savedSheetUrl);

          setIsConnected(true);

        }

        const savedNotifications = localStorage.getItem('notificationsEnabled');

        if (savedNotifications === 'true') setNotificationsEnabled(true);

      }, []);


      const getTodayKey = () => new Date().toISOString().split('T')[0];

      const handleValueChange = (attrId, value) => {

        setCurrentEntry(prev => ({ ...prev, [attrId]: value }));

      };


      const syncToGoogleSheet = async (entry) => {

        if (!googleSheetUrl) return;

        setIsSyncing(true);

        try {

          const allAttrs = [...attributes, ...customAttributes];

          const row = [entry.date, entry.timestamp, ...allAttrs.filter(a => a.enabled).map(a => {

            const val = entry[a.id];

            return val === true ? 'Yes' : val === false ? 'No' : val || '';

          })];

          await fetch(googleSheetUrl, {

            method: 'POST',

            mode: 'no-cors',

            headers: { 'Content-Type': 'application/json' },

            body: JSON.stringify({ values: [row] })

          });

        } catch (error) {

          console.error('Sync error:', error);

        } finally {

          setIsSyncing(false);

        }

      };


      const saveEntry = () => {

        const today = getTodayKey();

        const newEntry = { date: today, timestamp: new Date().toISOString(), ...currentEntry };

        const updated = entries.filter(e => e.date !== today);

        updated.push(newEntry);

        setEntries(updated);

        localStorage.setItem('wellnessEntries', JSON.stringify(updated));

        if (isConnected) syncToGoogleSheet(newEntry);

        setCurrentEntry({});

        alert('Entry saved successfully! 🎉' + (isConnected ? ' Synced to Google Sheets.' : ''));

      };


      const addCustomAttribute = () => {

        if (!newAttributeName.trim()) return;

        const newAttr = {

          id: `custom_${Date.now()}`,

          name: newAttributeName,

          type: newAttributeType,

          category: 'lifestyle',

          enabled: true,

          custom: true,

          ...(newAttributeType === 'scale' && { min: 1, max: 10 })

        };

        const updated = [...customAttributes, newAttr];

        setCustomAttributes(updated);

        localStorage.setItem('customAttributes', JSON.stringify(updated));

        setNewAttributeName('');

        setNewAttributeType('boolean');

      };


      const removeCustomAttribute = (id) => {

        const updated = customAttributes.filter(a => a.id !== id);

        setCustomAttributes(updated);

        localStorage.setItem('customAttributes', JSON.stringify(updated));

      };


      const toggleAttribute = (id) => {

        setAttributes(prev => {

          const updated = prev.map(a => a.id === id ? { ...a, enabled: !a.enabled } : a);

          const enabledState = {};

          updated.forEach(attr => { enabledState[attr.id] = attr.enabled; });

          localStorage.setItem('attributesEnabled', JSON.stringify(enabledState));

          return updated;

        });

      };


      const toggleCategory = (category) => {

        const updated = { ...expandedCategories, [category]: !expandedCategories[category] };

        setExpandedCategories(updated);

        localStorage.setItem('expandedCategories', JSON.stringify(updated));

      };


      const connectGoogleSheet = () => {

        if (googleSheetUrl.trim()) {

          localStorage.setItem('googleSheetUrl', googleSheetUrl);

          setIsConnected(true);

          alert('✅ Connected to Google Sheets! Your future entries will be synced automatically.');

        }

      };


      const disconnectGoogleSheet = () => {

        localStorage.removeItem('googleSheetUrl');

        setGoogleSheetUrl('');

        setIsConnected(false);

      };


      const requestNotificationPermission = async () => {

        if ('Notification' in window) {

          const permission = await Notification.requestPermission();

          if (permission === 'granted') {

            setNotificationsEnabled(true);

            localStorage.setItem('notificationsEnabled', 'true');

            alert('✅ Notifications enabled!');

          }

        } else {

          alert('❌ Notifications not supported');

        }

      };


      const exportData = (format) => {

        const allAttrs = [...attributes, ...customAttributes];

        if (format === 'json') {

          const data = { exportDate: new Date().toISOString(), version: '2.0', attributes: allAttrs, entries: entries };

          const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });

          downloadBlob(blob, `wellness-data-${getTodayKey()}.json`);

        } else if (format === 'csv') {

          const headers = ['Date', 'Timestamp', ...allAttrs.filter(a => a.enabled).map(a => a.name)];

          const rows = entries.map(entry => [entry.date, entry.timestamp, ...allAttrs.filter(a => a.enabled).map(a => {

            const val = entry[a.id];

            return val === true ? 'Yes' : val === false ? 'No' : val || '';

          })]);

          const csv = [headers, ...rows].map(row => row.join(',')).join('\n');

          const blob = new Blob([csv], { type: 'text/csv' });

          downloadBlob(blob, `wellness-data-${getTodayKey()}.csv`);

        }

      };


      const downloadBlob = (blob, filename) => {

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');

        a.href = url;

        a.download = filename;

        a.click();

        URL.revokeObjectURL(url);

      };


      const allAttrs = [...attributes, ...customAttributes].filter(a => a.enabled);

      const getCategoryName = (cat) => {

        const names = { sleep: 'Sleep & Rest', recovery: 'Recovery', physical: 'Physical Activity', mental: 'Mental Wellbeing', nutrition: 'Nutrition & Diet', supplements: 'Supplements', medications: 'Medications', lifestyle: 'Lifestyle & Health' };

        return names[cat] || cat;

      };


      const getCategoryIcon = (cat) => {

        const icons = { sleep: '😴', recovery: '💆', physical: '💪', mental: '🧠', nutrition: '🥗', supplements: '💊', medications: '💉', lifestyle: '🌍' };

        return icons[cat] || '📝';

      };


      const groupedAttributes = allAttrs.reduce((acc, attr) => {

        const cat = attr.category || 'lifestyle';

        if (!acc[cat]) acc[cat] = [];

        acc[cat].push(attr);

        return acc;

      }, {});


      const renderInput = (attr) => {

        const value = currentEntry[attr.id];

        if (attr.type === 'boolean') {

          return (

            <div className="flex gap-2">

              <button onClick={() => handleValueChange(attr.id, true)} className={`flex-1 py-3 rounded-lg font-medium transition-all ${value === true ? 'bg-green-500 text-white shadow-lg' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}`}>

                <Check className="inline w-5 h-5 mr-1" />Yes

              </button>

              <button onClick={() => handleValueChange(attr.id, false)} className={`flex-1 py-3 rounded-lg font-medium transition-all ${value === false ? 'bg-red-500 text-white shadow-lg' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}`}>

                <X className="inline w-5 h-5 mr-1" />No

              </button>

            </div>

          );

        }

        if (attr.type === 'scale') {

          return (

            <div>

              <input type="range" min={attr.min} max={attr.max} value={value || attr.min} onChange={(e) => handleValueChange(attr.id, parseInt(e.target.value))} className="w-full h-3 rounded-lg appearance-none cursor-pointer" style={{background: 'linear-gradient(to right, #fca5a5 0%, #fde047 50%, #86efac 100%)'}} />

              <div className="flex justify-between text-sm text-gray-600 mt-2">

                <span>Low {attr.min}</span>

                <span className="font-bold text-2xl text-blue-600">{value || attr.min}</span>

                <span>{attr.max} High</span>

              </div>

            </div>

          );

        }

        if (attr.type === 'number') {

          return <input type="number" value={value || ''} onChange={(e) => handleValueChange(attr.id, e.target.value)} className="w-full px-4 py-3 border-2 border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500" placeholder="Enter value" min="0" />;

        }

      };


      if (showSettings) {

        return (

          <div className="max-w-4xl mx-auto p-6 bg-white min-h-screen">

            <div className="mb-6 flex justify-between items-center">

              <h2 className="text-2xl font-bold text-gray-800">Settings</h2>

              <button onClick={() => setShowSettings(false)} className="px-6 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600">Done</button>

            </div>

            <div className="mb-8 p-6 bg-gradient-to-br from-green-50 to-blue-50 rounded-lg border-2 border-green-200">

              <h3 className="text-xl font-semibold mb-4 flex items-center gap-2">

                {isConnected ? <Cloud className="w-6 h-6 text-green-600" /> : <CloudOff className="w-6 h-6 text-gray-400" />}

                Google Sheets Integration

              </h3>

              {!isConnected ? (

                <div>

                  <p className="text-sm text-gray-600 mb-4">Connect to sync your data to Google Sheets</p>

                  <input type="text" value={googleSheetUrl} onChange={(e) => setGoogleSheetUrl(e.target.value)} placeholder="Paste Google Apps Script URL" className="w-full px-4 py-3 mb-3 border-2 rounded-lg" />

                  <button onClick={connectGoogleSheet} className="w-full py-3 bg-green-500 text-white rounded-lg hover:bg-green-600">Connect</button>

                </div>

              ) : (

                <div>

                  <div className="flex items-center gap-2 mb-3">

                    <div className="w-3 h-3 bg-green-500 rounded-full animate-pulse"></div>

                    <span className="text-green-700 font-medium">Connected</span>

                  </div>

                  <button onClick={disconnectGoogleSheet} className="py-2 px-4 bg-red-500 text-white rounded-lg">Disconnect</button>

                </div>

              )}

            </div>

            <div className="mb-8">

              <h3 className="text-xl font-semibold mb-4">Tracking Attributes ({allAttrs.length} enabled)</h3>

              {['sleep', 'recovery', 'physical', 'mental', 'nutrition', 'supplements', 'medications', 'lifestyle'].map(category => {

                const categoryAttrs = attributes.filter(a => a.category === category);

                if (categoryAttrs.length === 0) return null;

                return (

                  <div key={category} className="mb-4 border rounded-lg overflow-hidden">

                    <div onClick={() => toggleCategory(category)} className="bg-gray-100 px-4 py-3 font-semibold flex items-center gap-2 cursor-pointer hover:bg-gray-200">

                      <span>{getCategoryIcon(category)}</span>

                      <span>{getCategoryName(category)}</span>

                      <span className="text-sm text-gray-600 ml-auto mr-2">({categoryAttrs.filter(a => a.enabled).length}/{categoryAttrs.length})</span>

                      {expandedCategories[category] ? <ChevronUp className="w-5 h-5" /> : <ChevronDown className="w-5 h-5" />}

                    </div>

                    {expandedCategories[category] && categoryAttrs.map(attr => (

                      <div key={attr.id} className="flex items-center justify-between py-3 px-4 border-t bg-white">

                        <span className="text-gray-700">{attr.name}</span>

                        <button onClick={() => toggleAttribute(attr.id)} className={`px-4 py-1 rounded-full text-sm ${attr.enabled ? 'bg-blue-500 text-white' : 'bg-gray-200 text-gray-600'}`}>

                          {attr.enabled ? 'Enabled' : 'Disabled'}

                        </button>

                      </div>

                    ))}

                  </div>

                );

              })}

            </div>

            <div className="mb-8">

              <h3 className="text-xl font-semibold mb-4">Custom Attributes</h3>

              <div className="mb-4 p-4 bg-blue-50 rounded-lg">

                <input type="text" value={newAttributeName} onChange={(e) => setNewAttributeName(e.target.value)} placeholder="Attribute name" className="w-full px-4 py-3 mb-3 border-2 rounded-lg" />

                <select value={newAttributeType} onChange={(e) => setNewAttributeType(e.target.value)} className="w-full px-4 py-3 mb-3 border-2 rounded-lg">

                  <option value="boolean">Yes/No Toggle</option>

                  <option value="scale">Scale (1-10)</option>

                  <option value="number">Number Input</option>

                </select>

                <button onClick={addCustomAttribute} className="w-full py-3 bg-blue-500 text-white rounded-lg hover:bg-blue-600"><Plus className="inline w-5 h-5 mr-2" />Add</button>

              </div>

              {customAttributes.map(attr => (

                <div key={attr.id} className="flex items-center justify-between py-3 px-4 border rounded-lg mb-2">

                  <span>{attr.name} ({attr.type})</span>

                  <button onClick={() => removeCustomAttribute(attr.id)} className="px-3 py-2 bg-red-500 text-white rounded-lg"><Trash2 className="w-4 h-4" /></button>

                </div>

              ))}

            </div>

            <div>

              <h3 className="text-xl font-semibold mb-4">Export Data</h3>

              <div className="grid grid-cols-2 gap-3">

                <button onClick={() => exportData('json')} className="py-3 bg-green-500 text-white rounded-lg"><Download className="inline w-5 h-5 mr-2" />JSON</button>

                <button onClick={() => exportData('csv')} className="py-3 bg-green-500 text-white rounded-lg"><Download className="inline w-5 h-5 mr-2" />CSV</button>

              </div>

            </div>

          </div>

        );

      }


      if (view === 'history') {

        return (

          <div className="max-w-4xl mx-auto p-6 bg-white min-h-screen">

            <div className="mb-6 flex justify-between items-center">

              <h2 className="text-2xl font-bold">Entry History</h2>

              <button onClick={() => setView('today')} className="px-6 py-2 bg-blue-500 text-white rounded-lg">Back to Today</button>

            </div>

            {entries.length === 0 ? (

              <div className="text-center py-12">

                <Calendar className="w-16 h-16 text-gray-300 mx-auto mb-4" />

                <p className="text-gray-500">No entries yet</p>

              </div>

            ) : (

              <div className="space-y-4">

                {entries.sort((a, b) => b.date.localeCompare(a.date)).map(entry => (

                  <div key={entry.date} className="p-5 bg-gradient-to-br from-blue-50 to-purple-50 rounded-lg border shadow-sm">

                    <div className="font-bold text-lg mb-3">{new Date(entry.date + 'T00:00:00').toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' })}</div>

                    <div className="grid grid-cols-2 md:grid-cols-3 gap-3">

                      {allAttrs.map(attr => {

                        const val = entry[attr.id];

                        if (val === undefined || val === null || val === '') return null;

                        return (

                          <div key={attr.id} className="bg-white p-3 rounded-md shadow-sm">

                            <div className="text-xs text-gray-500 mb-1">{attr.name}</div>

                            <div className="font-semibold">{val === true ? '✅ Yes' : val === false ? '❌ No' : val}</div>

                          </div>

                        );

                      })}

                    </div>

                  </div>

                ))}

              </div>

            )}

          </div>

        );

      }


      return (

        <div className="max-w-4xl mx-auto p-6 bg-gradient-to-b from-blue-50 via-purple-50 to-white min-h-screen">

          <div className="mb-6 flex justify-between items-center flex-wrap gap-3">

            <h1 className="text-3xl md:text-4xl font-bold">Daily Wellness Tracker</h1>

            <div className="flex gap-2">

              <button onClick={() => setView('history')} className="px-4 py-2 bg-purple-500 text-white rounded-lg flex items-center gap-2">

                <Calendar className="w-5 h-5" />History

              </button>

              <button onClick={() => setShowSettings(true)} className="px-4 py-2 bg-gray-700 text-white rounded-lg flex items-center gap-2">

                <Settings className="w-5 h-5" />Settings

              </button>

            </div>

          </div>

          <div className="mb-4 p-3 bg-white rounded-lg shadow-sm border flex items-center justify-between">

            <div className="flex items-center gap-2">

              {isConnected ? (

                <><Cloud className="w-5 h-5 text-green-600" /><span className="text-sm text-green-700 font-medium">Synced to Google Sheets</span></>

              ) : (

                <><CloudOff className="w-5 h-5 text-gray-400" /><span className="text-sm text-gray-600">Local storage only</span></>

              )}

            </div>

            {isSyncing && <span className="text-xs text-blue-600 animate-pulse">Syncing...</span>}

          </div>

          <div className="mb-6 p-4 bg-white rounded-lg shadow-md border-l-4 border-blue-500">

            <p className="text-lg">📅 <strong>{new Date().toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' })}</strong></p>

          </div>

          <div className="space-y-6">

            {Object.entries(groupedAttributes).map(([category, attrs]) => (

              <div key={category} className="bg-white rounded-lg shadow-md overflow-hidden">

                <div className="bg-gradient-to-r from-blue-500 to-purple-500 px-5 py-3 flex items-center gap-2">

                  <span className="text-2xl">{getCategoryIcon(category)}</span>

                  <h2 className="text-xl font-bold text-white">{getCategoryName(category)}</h2>

                </div>

                <div className="p-5 space-y-4">

                  {attrs.map(attr => (

                    <div key={attr.id} className="border-b pb-4 last:border-b-0">

                      <label className="block text-gray-700 font-semibold mb-2">{attr.name}</label>

                      {renderInput(attr)}

                    </div>

                  ))}

                </div>

              </div>

            ))}

          </div>

          <div className="mt-8 sticky bottom-6">

            <button onClick={saveEntry} disabled={isSyncing} className="w-full py-4 bg-gradient-to-r from-green-500 to-blue-500 text-white rounded-lg font-bold text-lg shadow-lg hover:scale-105 transition-all disabled:opacity-50">

              {isSyncing ? '⏳ Syncing...' : '💾 Save Today\'s Entry'}

            </button>

          </div>

        </div>

      );

    };
